Summary
This component renders the teacher's class schedule within the EduTrack system, filtering entries to show only those active in the current academic term. It fetches schedule data and school-specific term configuration from the backend API upon mounting. The logic calculates the active term and validates each entry against grade-level end dates to ensure accuracy. It displays a loading spinner while fetching data and handles errors or empty states with appropriate user feedback. The interface uses Material UI components to present the schedule in a clean table format.

22/** Parse a class="str">"MM-dd" string into a Date for the given year. */
23class="cmt">// Converts a month-day string into a full JavaScript Date object
24const parseMMDD = (mmdd, year) => {
25 class="cmt">// Split the class="str">"MM-DD" string by hyphen and convert parts to integers
26 const [m, d] = mmdd.split(class="str">'-').map(Number);
27 class="cmt">// Return a new Date using the provided year, zero-indexed month, and day
28 return new Date(year, m - class="num">1, d);
29};
30
31/**
32 * Return the calendar year in which the current school year STARTED.
33 * School years straddle two calendar years (e.g. class="num">2024/class="num">2025 starts in class="num">2024).
34 */
35class="cmt">// Determine the start year of the current school year based on a given start date
36const schoolYearStartYear = (today, startDate) => {
37 class="cmt">// Parse the month and day from the YYYY-MM-DD string format
38 const [m, d] = startDate.split(class="str">'-').map(Number);
39 class="cmt">// Get the current calendar year
40 const y = today.getFullYear();
41 class="cmt">// If the current date has reached or passed the start date in the current year, return current year; otherwise, return the previous year
42 return today >= new Date(y, m - class="num">1, d) ? y : y - class="num">1;
43};
44
45/**
46 * Return class="num">1, class="num">2, or null (null = not in session).
47 * term2Start and end dates are always in startYear+class="num">1 (the class="str">"spring" half).
48 */
49const getActiveTerm = (today, cfg) => {
50 class="cmt">// Determine which academic year the current date belongs to
51 const sy = schoolYearStartYear(today, cfg.startDate);
52 class="cmt">// Calculate the start date of the first term for the current school year
53 const start = parseMMDD(cfg.startDate, sy);
54 class="cmt">// Calculate the start date of the second term for the following school year
55 const t2 = parseMMDD(cfg.term2Start, sy + class="num">1);
56 class="cmt">// If the current date is before the start of the first term, no term is active
57 if (today < start) return null;
58 class="cmt">// If the current date is before the start of the second term, we are in term class="num">1
59 if (today < t2) return class="num">1;
60 class="cmt">// Otherwise, the current date falls within the second term
61 return class="num">2;
62};
63
64/** True if the entry belongs to the active term and its grade level hasn't ended yet. */
65class="cmt">// Determines if a specific schedule entry is valid for the current academic term and date
66const isEntryActive = (entry, today, cfg, activeTerm) => {
67 class="cmt">// Validate that the entry belongs to the currently active school term
68 if (activeTerm === null || entry.term !== activeTerm) return false;
69
70 class="cmt">// Extract the numeric grade level from the class name (e.g., class="str">"class="num">5.A" -> class="num">5)
71 const grade = parseInt(entry.className.match(/^(\d+)/)?.[class="num">1] ?? class="str">'class="num">0');
72
73 class="cmt">// Determine the academic year end date based on the student's grade level category
74 const endStr = grade <= class="num">4 ? cfg.elementaryEnd
75 : grade <= class="num">7 ? cfg.progymnasiumEnd
76 : cfg.gymnasiumEnd;
77
78 class="cmt">// Calculate the start of the current school year
79 const sy = schoolYearStartYear(today, cfg.startDate);
80
81 class="cmt">// Parse the end date string using the calculated school year context
82 const end = parseMMDD(endStr, sy + class="num">1);
83
84 class="cmt">// The entry is active if the current date has not yet passed the term's end date
85 return today <= end;
86};

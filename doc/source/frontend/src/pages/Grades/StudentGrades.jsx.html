Summary
This component displays a student's academic performance within the EduTrack system by fetching and rendering grade data. It dynamically adjusts the API endpoint based on whether the current user is a student or an administrator. The interface allows users to filter grades by academic term using a tabbed navigation system. It calculates and displays subject-specific averages alongside an overall class average using color-coded chips for quick visual assessment. The main function handles data retrieval, state management, and the computation of statistical summaries for display.

12const GRADE_COLOR = v => {
13 class="cmt">// Map a numeric grade to a Material UI alert/text severity
14 const n = parseFloat(v);
15 class="cmt">// High grades (class="num">5.5 - class="num">6.0) use green success color
16 if (n >= class="num">5.5) return class="str">'success';
17 class="cmt">// Good grades (class="num">4.0 - class="num">5.0) use blue primary color
18 if (n >= class="num">4) return class="str">'primary';
19 class="cmt">// Passing/Borderline grades (class="num">2.0 - class="num">3.5) use orange warning color
20 if (n <= class="num">3) return class="str">'warning';
21 class="cmt">// Failing grades (below class="num">2.0) use red error color

25function StudentGrades() {
26 const { studentId } = useParams();
27 const { t } = useTranslation();
28 const { user } = useAuth();
29
30 const [grades, setGrades] = useState([]);
31 const [loading, setLoading] = useState(true);
32 const [error, setError] = useState(null);
33 const [termTab, setTermTab] = useState(class="num">0);
34
35 useEffect(() => {
36 const url = user?.role === class="str">'STUDENT'
37 ? class="str">'/api/grades/student/me'
38 : class="str">`/api/grades/student/${studentId}`;
39 api.get(url)
40 .then(res => setGrades(res.data))
41 .catch(() => setError(t(class="str">'grades.fetchError')))
42 .finally(() => setLoading(false));
43 }, [studentId]);
44
45 const termFilter = termTab === class="num">0 ? null : termTab;
46 const filtered = termFilter ? grades.filter(g => g.term === termFilter) : grades;
47
48 const avg = filtered.length
49 ? (filtered.reduce((s, g) => s + parseFloat(g.value), class="num">0) / filtered.length).toFixed(class="num">2)
50 : null;
51
52 const subjectAverages = useMemo(() => {
53 const map = {};
54 filtered.forEach(g => {
55 if (!map[g.subjectId]) map[g.subjectId] = { name: g.subjectName, sum: class="num">0, count: class="num">0 };
56 map[g.subjectId].sum += parseFloat(g.value);
57 map[g.subjectId].count += class="num">1;
58 });
59 return Object.values(map)
60 .map(s => ({ name: s.name, avg: (s.sum / s.count).toFixed(class="num">2) }))
61 .sort((a, b) => a.name.localeCompare(b.name));

63
64 return (
65 <Layout>

74 <>

94 ))}
95 </Box>
96 <Divider sx={{ mb: class="num">1.5 }} />
97 <Chip
98 label={class="str">`${t('grades.average')} (${t('grades.all')}): ${avg}`}
99 color={GRADE_COLOR(avg)}
100 />

102 )}
103
104 <TableContainer component={Paper}>
105 <Table size=class="str">"small">
106 <TableHead>
107 <TableRow>
108 <TableCell>{t(class="str">'grades.subject')}</TableCell>
109 <TableCell>{t(class="str">'grades.teacher')}</TableCell>
110 <TableCell>{t(class="str">'grades.term')}</TableCell>
111 <TableCell>{t(class="str">'grades.value')}</TableCell>
112 <TableCell>{t(class="str">'grades.date')}</TableCell>

115 <TableBody>
116 {filtered.length === class="num">0 ? (

118 <TableCell colSpan={class="num">5} align=class="str">"center">{t(class="str">'grades.noGrades')}</TableCell>
119 </TableRow>

122 <TableRow key={g.id} hover>
123 <TableCell>{g.subjectName}</TableCell>

125 <TableCell>{g.term}</TableCell>
126 <TableCell>

130 </TableRow>

132 )}
133 </TableBody>
134 </Table>
135 </TableContainer>
136 </>
137 )}
138 </Box>
139 </Layout>
140 );

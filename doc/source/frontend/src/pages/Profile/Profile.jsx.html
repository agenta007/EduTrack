Summary
This component manages the user profile page within the EduTrack system, allowing users to view their personal details and update their account settings. It handles profile picture uploads and deletions by interacting with the backend API and refreshing the avatar display. Users can edit and save a short biography, with the component providing immediate feedback on save success or failure. Additionally, it enables password changes by validating that new passwords match before submitting the request to the server. The interface uses Material UI components for a consistent look and supports internationalization for text labels.

5} from class="str">'class="ann">@mui/material';

18function Profile() {
19 const { t } = useTranslation();
20 const { user } = useAuth();
21
22 const fileInputRef = useRef(null);
23 const [avatarVersion, setAvatarVersion] = useState(class="num">0);
24 const [pictureError, setPictureError] = useState(null);
25
26 const [bio, setBio] = useState(class="str">'');
27 const [bioSaving, setBioSaving] = useState(false);
28 const [bioSuccess, setBioSuccess] = useState(false);
29 const [bioError, setBioError] = useState(null);
30
31 const [form, setForm] = useState({ currentPassword: class="str">'', newPassword: class="str">'', confirmPassword: class="str">'' });
32 const [saving, setSaving] = useState(false);
33 const [success, setSuccess] = useState(false);
34 const [error, setError] = useState(null);
35
36 useEffect(() => {
37 api.get(class="str">'/api/profile').then(res => setBio(res.data.bio ?? class="str">''));
38 }, []);
39
40 const handlePictureChange = (e) => {
41 const file = e.target.files[class="num">0];
42 if (!file) return;
43 e.target.value = class="str">'';
44 setPictureError(null);
45
46 const formData = new FormData();
47 formData.append(class="str">'file', file);
48 api.put(class="str">'/api/profile/picture', formData, { headers: { class="str">'Content-Type': class="str">'multipart/form-data' } })
49 .then(() => setAvatarVersion(v => v + class="num">1))
50 .catch(() => setPictureError(t(class="str">'profile.pictureError')));
51 };
52
53 const handleBioSave = () => {
54 setBioError(null);
55 setBioSuccess(false);
56 setBioSaving(true);
57 api.put(class="str">'/api/profile/bio', { bio })
58 .then(() => setBioSuccess(true))
59 .catch(() => setBioError(t(class="str">'profile.bioError')))
60 .finally(() => setBioSaving(false));
61 };
62
63 const handleSubmit = () => {
64 setError(null);
65 setSuccess(false);
66
67 if (form.newPassword !== form.confirmPassword) {
68 setError(t(class="str">'profile.passwordMismatch'));
69 return;
70 }
71
72 setSaving(true);
73 api.put(class="str">'/api/profile/password', {
74 currentPassword: form.currentPassword,
75 newPassword: form.newPassword,
76 })
77 .then(() => {
78 setSuccess(true);
79 setForm({ currentPassword: class="str">'', newPassword: class="str">'', confirmPassword: class="str">'' });
80 })
81 .catch(err => setError(err.response?.data?.message || t(class="str">'profile.changeError')))
82 .finally(() => setSaving(false));
83 };
84
85 return (
86 <Layout>
87 <Box sx={{ p: class="num">3, maxWidth: class="num">500 }}>
88 <Typography variant=class="str">"h5" sx={{ mb: class="num">3 }}>{t(class="str">'profile.title')}</Typography>
89
90 {/* User info */}
91 <Card sx={{ mb: class="num">3 }}>
92 <CardContent>

96 <Box
97 sx={{ position: class="str">'relative', cursor: class="str">'pointer' }}
98 onClick={() => fileInputRef.current.click()}
99 title={t(class="str">'profile.uploadPicture')}
100 >
101 <UserAvatar userId={user?.id} name={user?.name} size={class="num">72} refreshToken={avatarVersion} />
102 <Box sx={{
103 position: class="str">'absolute', bottom: class="num">0, right: class="num">0,
104 bgcolor: class="str">'primary.main', borderRadius: class="str">'class="num">50%',
105 width: class="num">22, height: class="num">22,
106 display: class="str">'flex', alignItems: class="str">'center', justifyContent: class="str">'center',
107 }}>
108 <CameraAltIcon sx={{ fontSize: class="num">13, color: class="str">'white' }} />
109 </Box>
110 </Box>
111 <Button
112 size=class="str">"small"
113 color=class="str">"error"
114 variant=class="str">"outlined"
115 sx={{ fontSize: class="num">11 }}
116 onClick={() => {
117 setPictureError(null);
118 api.delete(class="str">'/api/profile/picture')
119 .then(() => setAvatarVersion(v => v + class="num">1))
120 .catch(() => setPictureError(t(class="str">'profile.pictureError')));
121 }}
122 >
123 {t(class="str">'profile.deletePicture')}
124 </Button>
125 </Box>
126
127 <Box>
128 <Typography><strong>{t(class="str">'users.firstName')}:</strong> {user?.name?.split(class="str">' ')[class="num">0]}</Typography>
129 <Typography><strong>{t(class="str">'users.lastName')}:</strong> {user?.name?.split(class="str">' ').slice(class="num">1).join(class="str">' ')}</Typography>
130 <Typography><strong>{t(class="str">'users.email')}:</strong> {user?.email}</Typography>
131 <Typography><strong>{t(class="str">'users.role')}:</strong> {user?.role}</Typography>
132 </Box>
133 </Box>
134
135 {pictureError && <Alert severity=class="str">"error" sx={{ mt: class="num">1 }}>{pictureError}</Alert>}
136 </CardContent>
137 </Card>
138
139 <input
140 type=class="str">"file"
141 accept=class="str">"image/*"
142 ref={fileInputRef}
143 style={{ display: class="str">'none' }}
144 onChange={handlePictureChange}
145 />
146
147 <Divider sx={{ mb: class="num">3 }} />
148
149 {/* Bio */}
150 <Typography variant=class="str">"h6" sx={{ mb: class="num">2 }}>{t(class="str">'profile.bio')}</Typography>
151
152 {bioSuccess && <Alert severity=class="str">"success" sx={{ mb: class="num">2 }}>{t(class="str">'profile.bioSaved')}</Alert>}
153 {bioError && <Alert severity=class="str">"error" sx={{ mb: class="num">2 }}>{bioError}</Alert>}
154
155 <Box sx={{ display: class="str">'flex', flexDirection: class="str">'column', gap: class="num">2, mb: class="num">3 }}>
156 <TextField
157 placeholder={t(class="str">'profile.bioPlaceholder')}
158 value={bio}
159 onChange={e => { setBio(e.target.value.slice(class="num">0, class="num">500)); setBioSuccess(false); }}
160 multiline
161 rows={class="num">4}
162 fullWidth
163 helperText={class="str">`${bio.length} / class="num">500`}
164 {...fieldSx}
165 />
166 <Button
167 variant=class="str">"contained"
168 onClick={handleBioSave}
169 disabled={bioSaving}
170 sx={{ alignSelf: class="str">'flex-start' }}
171 >
172 {t(class="str">'common.save')}
173 </Button>
174 </Box>
175
176 <Divider sx={{ mb: class="num">3 }} />
177
178 {/* Change password */}
179 <Typography variant=class="str">"h6" sx={{ mb: class="num">2 }}>{t(class="str">'profile.changePassword')}</Typography>

181 {success && <Alert severity=class="str">"success" sx={{ mb: class="num">2 }}>{t(class="str">'profile.changeSuccess')}</Alert>}
182 {error && <Alert severity=class="str">"error" sx={{ mb: class="num">2 }}>{error}</Alert>}
183
184 <Box sx={{ display: class="str">'flex', flexDirection: class="str">'column', gap: class="num">2 }}>
185 <TextField
186 label={t(class="str">'profile.currentPassword')}
187 type=class="str">"password"
188 value={form.currentPassword}
189 onChange={e => setForm(f => ({ ...f, currentPassword: e.target.value }))}
190 fullWidth
191 {...fieldSx}
192 />
193 <TextField
194 label={t(class="str">'profile.newPassword')}
195 type=class="str">"password"
196 value={form.newPassword}
197 onChange={e => setForm(f => ({ ...f, newPassword: e.target.value }))}
198 fullWidth
199 {...fieldSx}
200 />
201 <TextField
202 label={t(class="str">'profile.confirmPassword')}
203 type=class="str">"password"
204 value={form.confirmPassword}

207 {...fieldSx}
208 />
209 <Button
210 variant=class="str">"contained"
211 onClick={handleSubmit}
212 disabled={!form.currentPassword || !form.newPassword || !form.confirmPassword || saving}
213 >
214 {t(class="str">'profile.changePassword')}
215 </Button>
216 </Box>
217 </Box>
218 </Layout>
219 );
220}
221
222export default Profile;

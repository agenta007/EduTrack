Summary
This component manages the display and creation of disciplinary complaints for a specific class within the EduTrack system. It fetches class details, existing complaints, student lists, and schedule data upon mounting, then renders them in a tabular format filtered by academic term. Users with administrative or teaching roles can add new complaints via a dialog form that dynamically filters available schedules based on the selected date. The interface supports internationalization for day labels and includes error handling for API interactions. It serves as the primary interface for class-level behavioral record keeping.

21/** Returns the Monday–Friday dates of the week containing class="str">`date`. */
22function getWeekDays(date) {
23 class="cmt">// Initialize a new date object from the provided input to avoid mutating the original
24 const d = new Date(date);
25 class="cmt">// Retrieve the day of the week (class="num">0 for Sunday, class="num">1 for Monday, etc.)
26 const day = d.getDay(); class="cmt">// class="num">0=Sun…class="num">6=Sat
27 class="cmt">// Create a reference date for the start of the week
28 const monday = new Date(d);
29 class="cmt">// Calculate and set the date to the most recent Monday
30 monday.setDate(d.getDate() - (day === class="num">0 ? class="num">6 : day - class="num">1));
31 class="cmt">// Generate an array representing the first class="num">5 weekdays (Monday to Friday)
32 return Array.from({ length: class="num">5 }, (_, i) => {
33 class="cmt">// Clone the Monday reference date for each iteration
34 const dd = new Date(monday);
35 class="cmt">// Offset the date by the current index to traverse the week
36 dd.setDate(monday.getDate() + i);
37 return dd;
38 });
39}

48/** day-of-week index the DB uses: Mon=class="num">1…Fri=class="num">5 */
49function dbDayOfWeek(date) {
50 class="cmt">// Convert JavaScript class="num">0-class="num">6 (Sun-Sat) to database-compatible class="num">1-class="num">7 (Mon-Sun) format
51 const js = date.getDay(); class="cmt">// class="num">0=Sun…class="num">6=Sat
52 class="cmt">// Map Sunday from class="num">0 to class="num">7, leaving other days unchanged
53 return js === class="num">0 ? class="num">7 : js; class="cmt">// Sun→class="num">7, Mon→class="num">1…Fri→class="num">5 (only class="num">1-class="num">5 are valid)
54}

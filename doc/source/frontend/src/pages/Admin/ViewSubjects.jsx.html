Summary
This component manages the administrative interface for viewing, creating, updating, and deleting subjects within the EduTrack system. It fetches subject data from the backend API and displays it in a table, restricting edit and delete actions to users with the admin role. The file includes logic for handling CRUD operations through modal dialogs for adding and editing subjects, as well as a confirmation dialog for deletions. It utilizes React state to manage loading, error, and form data states while ensuring a responsive user experience with appropriate feedback messages.

38function ViewSubjects() {
39 const { t } = useTranslation();
40 const { user } = useAuth();
41 const canEdit = user?.role === class="str">'ADMIN';
42
43 const [subjects, setSubjects] = useState([]);
44 const [loading, setLoading] = useState(true);
45 const [error, setError] = useState(null);
46 const [saving, setSaving] = useState(false);
47
48 class="cmt">// Add dialog
49 const [addOpen, setAddOpen] = useState(false);
50 const [addForm, setAddForm] = useState(emptyForm);
51
52 class="cmt">// Edit dialog
53 const [editOpen, setEditOpen] = useState(false);
54 const [editId, setEditId] = useState(null);
55 const [editForm, setEditForm] = useState(emptyForm);
56
57 class="cmt">// Delete confirm
58 const [confirmId, setConfirmId] = useState(null);
59 const [confirmName, setConfirmName] = useState(class="str">'');
60
61 const fetchSubjects = () => {
62 setLoading(true);
63 setError(null);
64 api.get(class="str">'/api/subjects')
65 .then(res => setSubjects(res.data))
66 .catch(() => setError(t(class="str">'subjects.fetchError')))
67 .finally(() => setLoading(false));
68 };
69
70 useEffect(() => { fetchSubjects(); }, []);
71
72 const handleCreate = () => {
73 setSaving(true);
74 api.post(class="str">'/api/subjects', { name: addForm.name })
75 .then(res => {
76 setSubjects(prev => [...prev, res.data]);
77 setAddOpen(false);
78 setAddForm(emptyForm);
79 })
80 .catch(() => setError(t(class="str">'subjects.createError')))
81 .finally(() => setSaving(false));
82 };
83
84 const handleRowClick = (subject) => {
85 setEditId(subject.id);
86 setEditForm({ name: subject.name });
87 setEditOpen(true);
88 };
89
90 const handleUpdate = () => {
91 setSaving(true);
92 api.put(class="str">`/api/subjects/${editId}`, { name: editForm.name })
93 .then(res => {
94 setSubjects(prev => prev.map(s => s.id === editId ? res.data : s));
95 setEditOpen(false);
96 })
97 .catch(() => setError(t(class="str">'subjects.updateError')))
98 .finally(() => setSaving(false));
99 };
100
101 const handleDeleteConfirm = () => {
102 api.delete(class="str">`/api/subjects/${confirmId}`)
103 .then(() => {
104 setSubjects(prev => prev.filter(s => s.id !== confirmId));
105 setConfirmId(null);
106 })
107 .catch(() => {
108 setError(t(class="str">'subjects.deleteError'));
109 setConfirmId(null);
110 });
111 };
112
113 return (
114 <Layout>
115 <Box sx={{ p: class="num">3 }}>
116 <Box sx={{ display: class="str">'flex', justifyContent: class="str">'space-between', alignItems: class="str">'center', mb: class="num">2 }}>
117 <Typography variant=class="str">"h5">{t(class="str">'subjects.title')}</Typography>
118 {canEdit && (
119 <Button variant=class="str">"contained" startIcon={<AddIcon />} onClick={() => setAddOpen(true)}>
120 {t(class="str">'subjects.addSubject')}
121 </Button>
122 )}
123 </Box>
124
125 {error && <Alert severity=class="str">"error" sx={{ mb: class="num">2 }}>{error}</Alert>}
126
127 {loading ? (
128 <Box sx={{ display: class="str">'flex', justifyContent: class="str">'center', mt: class="num">4 }}>
129 <CircularProgress />
130 </Box>
131 ) : (
132 <TableContainer component={Paper}>
133 <Table>
134 <TableHead>
135 <TableRow>
136 <TableCell>ID</TableCell>
137 <TableCell>{t(class="str">'subjects.name')}</TableCell>
138 {canEdit && <TableCell align=class="str">"center">{t(class="str">'subjects.actions')}</TableCell>}
139 </TableRow>
140 </TableHead>
141 <TableBody>
142 {subjects.length === class="num">0 ? (
143 <TableRow>
144 <TableCell colSpan={canEdit ? class="num">3 : class="num">2} align=class="str">"center">{t(class="str">'subjects.noSubjects')}</TableCell>
145 </TableRow>
146 ) : (
147 subjects.map(subject => (
148 <TableRow
149 key={subject.id}
150 hover
151 onClick={() => canEdit && handleRowClick(subject)}
152 sx={{ cursor: canEdit ? class="str">'pointer' : class="str">'default' }}
153 >
154 <TableCell>{subject.id}</TableCell>
155 <TableCell>{subject.name}</TableCell>
156 {canEdit && (
157 <TableCell align=class="str">"center">
158 <IconButton onClick={e => { e.stopPropagation(); setConfirmId(subject.id); setConfirmName(subject.name); }}>
159 <DeleteIcon sx={{ color: class="str">'red' }} />
160 </IconButton>
161 </TableCell>
162 )}
163 </TableRow>
164 ))
165 )}
166 </TableBody>
167 </Table>
168 </TableContainer>
169 )}
170 </Box>
171
172 {/* Add Subject Dialog */}
173 <Dialog open={addOpen} onClose={() => setAddOpen(false)} fullWidth maxWidth=class="str">"sm">
174 <DialogTitle>{t(class="str">'subjects.addSubject')}</DialogTitle>
175 <DialogContent sx={{ display: class="str">'flex', flexDirection: class="str">'column', gap: class="num">2, pt: class="num">2 }}>
176 <SubjectFormFields form={addForm} setForm={setAddForm} t={t} />
177 </DialogContent>
178 <DialogActions>
179 <Button onClick={() => setAddOpen(false)}>{t(class="str">'common.cancel')}</Button>
180 <Button variant=class="str">"contained" onClick={handleCreate} disabled={!addForm.name.trim() || saving}>
181 {t(class="str">'common.save')}
182 </Button>
183 </DialogActions>
184 </Dialog>
185
186 {/* Edit Subject Dialog */}
187 <Dialog open={editOpen} onClose={() => setEditOpen(false)} fullWidth maxWidth=class="str">"sm">
188 <DialogTitle>{t(class="str">'subjects.editSubject')}</DialogTitle>
189 <DialogContent sx={{ display: class="str">'flex', flexDirection: class="str">'column', gap: class="num">2, pt: class="num">2 }}>
190 <SubjectFormFields form={editForm} setForm={setEditForm} t={t} />
191 </DialogContent>
192 <DialogActions>
193 <Button onClick={() => setEditOpen(false)}>{t(class="str">'common.cancel')}</Button>
194 <Button variant=class="str">"contained" onClick={handleUpdate} disabled={!editForm.name.trim() || saving}>
195 {t(class="str">'common.save')}
196 </Button>
197 </DialogActions>
198 </Dialog>
199
200 {/* Delete Confirmation Dialog */}
201 <Dialog open={confirmId !== null} onClose={() => setConfirmId(null)}>
202 <DialogTitle>{t(class="str">'subjects.deleteConfirmTitle')}</DialogTitle>
203 <DialogContent>
204 <Typography>{t(class="str">'subjects.deleteConfirmMessage', { name: confirmName })}</Typography>
205 </DialogContent>
206 <DialogActions>
207 <Button onClick={() => setConfirmId(null)}>{t(class="str">'common.cancel')}</Button>
208 <Button variant=class="str">"contained" color=class="str">"error" onClick={handleDeleteConfirm}>
209 {t(class="str">'common.delete')}
210 </Button>
211 </DialogActions>
212 </Dialog>
213 </Layout>
214 );
215}
216
217export default ViewSubjects;

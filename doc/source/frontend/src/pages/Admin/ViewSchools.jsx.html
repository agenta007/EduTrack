Summary
This file implements the administrative interface for managing school data within the EduTrack system. It provides a form for creating and editing school details such as name, address, type, and headmaster assignment. The component includes a ProfileManager to handle school-specific user profiles and a ScheduleManager for organizing daily academic entries like lectures and breaks. It utilizes Material UI components for a consistent user experience and integrates with the backend API for all data persistence operations. The module supports internationalization to ensure accessibility across different languages.

71function ProfileManager({ schoolId, t }) {
72 const [profiles, setProfiles] = useState([]);
73 const [newName, setNewName] = useState(class="str">'');
74 const [loading, setLoading] = useState(true);
75
76 useEffect(() => {
77 api.get(class="str">`/api/schools/${schoolId}/profiles`)
78 .then(res => setProfiles(res.data))
79 .finally(() => setLoading(false));
80 }, [schoolId]);
81
82 const handleAdd = () => {
83 if (!newName.trim()) return;
84 api.post(class="str">`/api/schools/${schoolId}/profiles`, { name: newName.trim() })
85 .then(res => { setProfiles(prev => [...prev, res.data]); setNewName(class="str">''); });
86 };
87
88 const handleDelete = (id) => {
89 api.delete(class="str">`/api/schools/profiles/${id}`)
90 .then(() => setProfiles(prev => prev.filter(p => p.id !== id)));
91 };
92
93 if (loading) return null;
94
95 return (
96 <Box>
97 <Divider sx={{ my: class="num">2 }} />
98 <Typography variant=class="str">"subtitle1" sx={{ mb: class="num">1 }}>{t(class="str">'schools.profiles')}</Typography>
99 <List dense disablePadding>
100 {profiles.length === class="num">0 && (
101 <Typography variant=class="str">"body2" color=class="str">"text.secondary" sx={{ mb: class="num">1 }}>
102 {t(class="str">'schools.noProfiles')}
103 </Typography>
104 )}
105 {profiles.map(p => (
106 <ListItem key={p.id} disablePadding sx={{ mb: class="num">0.5 }}
107 secondaryAction={
108 <IconButton edge=class="str">"end" size=class="str">"small" onClick={() => handleDelete(p.id)}>
109 <DeleteIcon fontSize=class="str">"small" sx={{ color: class="str">'red' }} />
110 </IconButton>
111 }
112 >
113 <ListItemText primary={p.name} />
114 </ListItem>
115 ))}
116 </List>
117 <Box sx={{ display: class="str">'flex', gap: class="num">1, mt: class="num">1 }}>
118 <TextField
119 size=class="str">"small"
120 label={t(class="str">'schools.addProfile')}
121 value={newName}
122 onChange={e => setNewName(e.target.value)}
123 onKeyDown={e => e.key === class="str">'Enter' && handleAdd()}
124 fullWidth
125 {...fieldProps}
126 />
127 <Button variant=class="str">"outlined" onClick={handleAdd} disabled={!newName.trim()}>
128 <AddIcon />
129 </Button>
130 </Box>
131 </Box>
132 );
133}

135function ScheduleManager({ schoolId, t }) {
136 const [entries, setEntries] = useState([]);
137 const [loading, setLoading] = useState(true);
138 const [showForm, setShowForm] = useState(false);
139 const [form, setForm] = useState(emptyEntry);
140 const [editingId, setEditingId] = useState(null);
141
142 useEffect(() => {
143 api.get(class="str">`/api/schools/${schoolId}/schedule`)
144 .then(res => setEntries(res.data))
145 .finally(() => setLoading(false));
146 }, [schoolId]);
147
148 const openAdd = () => { setEditingId(null); setForm(emptyEntry); setShowForm(true); };
149 const openEdit = (e) => {
150 setEditingId(e.id);
151 setForm({ type: e.type, label: e.label, startTime: e.startTime, endTime: e.endTime, eventDate: e.eventDate || class="str">'' });
152 setShowForm(true);
153 };
154 const closeForm = () => { setShowForm(false); setEditingId(null); setForm(emptyEntry); };
155
156 const handleSave = () => {
157 if (!form.label.trim() || !form.startTime || !form.endTime) return;
158 const req = editingId
159 ? api.put(class="str">`/api/schools/schedule/${editingId}`, form)
160 : api.post(class="str">`/api/schools/${schoolId}/schedule`, form);
161 req.then(res => {
162 setEntries(prev =>
163 editingId
164 ? prev.map(e => e.id === editingId ? res.data : e)
165 : [...prev, res.data]
166 );
167 closeForm();
168 });
169 };
170
171 const handleDelete = (id) => {
172 api.delete(class="str">`/api/schools/schedule/${id}`)
173 .then(() => setEntries(prev => prev.filter(e => e.id !== id)));
174 };
175
176 if (loading) return null;
177
178 return (
179 <Box>
180 <Divider sx={{ my: class="num">2 }} />
181 <Box sx={{ display: class="str">'flex', justifyContent: class="str">'space-between', alignItems: class="str">'center', mb: class="num">1 }}>
182 <Typography variant=class="str">"subtitle1">{t(class="str">'schools.dailySchedule')}</Typography>
183 <Button size=class="str">"small" startIcon={<AddIcon />} onClick={openAdd}>
184 {t(class="str">'schools.addScheduleEntry')}
185 </Button>
186 </Box>
187
188 {entries.length === class="num">0 && !showForm && (
189 <Typography variant=class="str">"body2" color=class="str">"text.secondary" sx={{ mb: class="num">1 }}>
190 {t(class="str">'schools.noScheduleEntries')}
191 </Typography>
192 )}
193
194 <List dense disablePadding>
195 {entries.map(e => (
196 <ListItem
197 key={e.id} disablePadding sx={{ mb: class="num">0.5, cursor: class="str">'pointer' }}
198 onClick={() => openEdit(e)}
199 secondaryAction={
200 <IconButton edge=class="str">"end" size=class="str">"small" onClick={ev => { ev.stopPropagation(); handleDelete(e.id); }}>
201 <DeleteIcon fontSize=class="str">"small" sx={{ color: class="str">'red' }} />
202 </IconButton>
203 }
204 >
205 <Box sx={{ display: class="str">'flex', alignItems: class="str">'center', gap: class="num">1, flex: class="num">1, pr: class="num">5, py: class="num">0.5 }}>
206 <Chip
207 label={t(class="str">`entryTypes.${e.type}`)}
208 size=class="str">"small"
209 color={ENTRY_TYPE_COLORS[e.type]}
210 sx={{ minWidth: class="num">90 }}
211 />
212 <Typography variant=class="str">"body2" sx={{ fontWeight: class="num">500, flexShrink: class="num">0 }}>{e.label}</Typography>
213 <Typography variant=class="str">"body2" color=class="str">"text.secondary">
214 {e.startTime}–{e.endTime}
215 {e.eventDate ? class="str">` · ${e.eventDate}` : class="str">''}
216 </Typography>
217 </Box>
218 </ListItem>
219 ))}
220 </List>
221
222 {showForm && (
223 <Box sx={{ mt: class="num">1, p: class="num">1.5, border: class="str">'1px solid', borderColor: class="str">'divider', borderRadius: class="num">1, display: class="str">'flex', flexDirection: class="str">'column', gap: class="num">1.5 }}>
224 <FormControl size=class="str">"small" fullWidth>
225 <InputLabel sx={{ color: class="str">'black', '&.Mui-focusedclass="str">': { color: 'black' } }}>{t(class="str">'schools.entryType')}</InputLabel>
226 <Select
227 value={form.type}
228 onChange={e => setForm(f => ({ ...f, type: e.target.value }))}
229 label={t(class="str">'schools.entryType')} sx={{ color: class="str">'black' }}
230 >
231 {ENTRY_TYPES.map(type => (
232 <MenuItem key={type} value={type}>{t(class="str">`entryTypes.${type}`)}</MenuItem>
233 ))}
234 </Select>
235 </FormControl>
236
237 <TextField
238 size=class="str">"small" label={t(class="str">'schools.entryLabel')} value={form.label}
239 onChange={e => setForm(f => ({ ...f, label: e.target.value }))}
240 fullWidth {...fieldProps}
241 />
242
243 <Box sx={{ display: class="str">'flex', gap: class="num">1 }}>
244 <TextField
245 size=class="str">"small" label={t(class="str">'schools.startTime')} type=class="str">"time" value={form.startTime}
246 onChange={e => setForm(f => ({ ...f, startTime: e.target.value }))}
247 fullWidth {...fieldProps}
248 />
249 <TextField
250 size=class="str">"small" label={t(class="str">'schools.endTime')} type=class="str">"time" value={form.endTime}
251 onChange={e => setForm(f => ({ ...f, endTime: e.target.value }))}
252 fullWidth {...fieldProps}
253 />
254 </Box>
255
256 {form.type === class="str">'SPECIAL_EVENT' && (
257 <TextField
258 size=class="str">"small" label={t(class="str">'schools.eventDate')} type=class="str">"date" value={form.eventDate}
259 onChange={e => setForm(f => ({ ...f, eventDate: e.target.value }))}
260 fullWidth {...fieldProps}
261 />
262 )}
263
264 <Box sx={{ display: class="str">'flex', gap: class="num">1, justifyContent: class="str">'flex-end' }}>
265 <Button size=class="str">"small" onClick={closeForm}>{t(class="str">'common.cancel')}</Button>
266 <Button
267 size=class="str">"small" variant=class="str">"contained"
268 disabled={!form.label.trim() || !form.startTime || !form.endTime}
269 onClick={handleSave}
270 >
271 {t(class="str">'common.save')}
272 </Button>
273 </Box>
274 </Box>
275 )}
276 </Box>
277 );
278}

280function ViewSchools() {
281 const { t } = useTranslation();
282
283 const [schools, setSchools] = useState([]);
284 const [loading, setLoading] = useState(true);
285 const [error, setError] = useState(null);
286 const [saving, setSaving] = useState(false);
287 const [headmasters, setHeadmasters] = useState([]);
288
289 const [addOpen, setAddOpen] = useState(false);
290 const [addForm, setAddForm] = useState(emptyForm);
291
292 const [editOpen, setEditOpen] = useState(false);
293 const [editId, setEditId] = useState(null);
294 const [editForm, setEditForm] = useState(emptyForm);
295
296 const [confirmId, setConfirmId] = useState(null);
297 const [confirmName, setConfirmName] = useState(class="str">'');
298
299 const fetchSchools = () => {
300 setLoading(true);
301 setError(null);
302 api.get(class="str">'/api/schools')
303 .then(res => setSchools(res.data))
304 .catch(() => setError(t(class="str">'schools.fetchError')))
305 .finally(() => setLoading(false));
306 };
307
308 useEffect(() => {
309 fetchSchools();
310 api.get(class="str">'/api/users/headmasters').then(res => setHeadmasters(res.data)).catch(() => {});
311 }, []);
312
313 const toPayload = form => ({
314 name: form.name,
315 address: form.address,
316 type: form.type || null,
317 headmasterId: form.headmasterId ? Number(form.headmasterId) : null,
318 });
319
320 const handleCreate = () => {
321 setSaving(true);
322 api.post(class="str">'/api/schools', toPayload(addForm))
323 .then(res => { setSchools(prev => [...prev, res.data]); setAddOpen(false); setAddForm(emptyForm); })
324 .catch(() => setError(t(class="str">'schools.createError')))
325 .finally(() => setSaving(false));
326 };
327
328 const handleRowClick = (school) => {
329 setEditId(school.id);
330 const match = headmasters.find(h => h.name === school.headmasterName);
331 setEditForm({
332 name: school.name || class="str">'',
333 address: school.address || class="str">'',
334 type: school.type || class="str">'',
335 headmasterId: match ? match.id : class="str">'',
336 });
337 setEditOpen(true);
338 };
339
340 const handleUpdate = () => {
341 setSaving(true);
342 api.put(class="str">`/api/schools/${editId}`, toPayload(editForm))
343 .then(res => { setSchools(prev => prev.map(s => s.id === editId ? res.data : s)); setEditOpen(false); })
344 .catch(() => setError(t(class="str">'schools.updateError')))
345 .finally(() => setSaving(false));
346 };
347
348 const handleDeleteConfirm = () => {
349 api.delete(class="str">`/api/schools/${confirmId}`)
350 .then(() => { setSchools(prev => prev.filter(s => s.id !== confirmId)); setConfirmId(null); })
351 .catch(() => { setError(t(class="str">'schools.deleteError')); setConfirmId(null); });
352 };
353
354 return (
355 <Layout>
356 <Box sx={{ p: class="num">3 }}>
357 <Box sx={{ display: class="str">'flex', justifyContent: class="str">'space-between', alignItems: class="str">'center', mb: class="num">2 }}>
358 <Typography variant=class="str">"h5">{t(class="str">'schools.title')}</Typography>
359 <Button variant=class="str">"contained" startIcon={<AddIcon />} onClick={() => setAddOpen(true)}>
360 {t(class="str">'schools.addSchool')}
361 </Button>
362 </Box>
363
364 {error && <Alert severity=class="str">"error" sx={{ mb: class="num">2 }}>{error}</Alert>}
365
366 {loading ? (
367 <Box sx={{ display: class="str">'flex', justifyContent: class="str">'center', mt: class="num">4 }}><CircularProgress /></Box>
368 ) : (
369 <TableContainer component={Paper}>
370 <Table>
371 <TableHead>
372 <TableRow>
373 <TableCell>ID</TableCell>
374 <TableCell>{t(class="str">'schools.name')}</TableCell>
375 <TableCell>{t(class="str">'schools.type')}</TableCell>
376 <TableCell>{t(class="str">'schools.headmaster')}</TableCell>
377 <TableCell>{t(class="str">'schools.profiles')}</TableCell>
378 <TableCell align=class="str">"center">{t(class="str">'schools.actions')}</TableCell>
379 </TableRow>
380 </TableHead>
381 <TableBody>
382 {schools.length === class="num">0 ? (
383 <TableRow>
384 <TableCell colSpan={class="num">6} align=class="str">"center">{t(class="str">'schools.noSchools')}</TableCell>
385 </TableRow>
386 ) : (
387 schools.map(school => (
388 <TableRow key={school.id} hover onClick={() => handleRowClick(school)} sx={{ cursor: class="str">'pointer' }}>
389 <TableCell>{school.id}</TableCell>
390 <TableCell>{school.name}</TableCell>
391 <TableCell>
392 {school.type
393 ? <Chip label={t(class="str">`schoolTypes.${school.type}`)} size=class="str">"small" />
394 : class="str">'—'}
395 </TableCell>
396 <TableCell>{school.headmasterName || class="str">'—'}</TableCell>
397 <TableCell>
398 <Box sx={{ display: class="str">'flex', gap: class="num">0.5, flexWrap: class="str">'wrap' }}>
399 {school.profiles?.length > class="num">0
400 ? school.profiles.map(p => <Chip key={p.id} label={p.name} size=class="str">"small" variant=class="str">"outlined" />)
401 : class="str">'—'}
402 </Box>
403 </TableCell>
404 <TableCell align=class="str">"center">
405 <IconButton onClick={e => { e.stopPropagation(); setConfirmId(school.id); setConfirmName(school.name); }}>
406 <DeleteIcon sx={{ color: class="str">'red' }} />
407 </IconButton>
408 </TableCell>
409 </TableRow>
410 ))
411 )}
412 </TableBody>
413 </Table>
414 </TableContainer>
415 )}
416 </Box>
417
418 {/* Add School Dialog */}
419 <Dialog open={addOpen} onClose={() => setAddOpen(false)} fullWidth maxWidth=class="str">"sm">
420 <DialogTitle>{t(class="str">'schools.addSchool')}</DialogTitle>
421 <DialogContent sx={{ display: class="str">'flex', flexDirection: class="str">'column', gap: class="num">2, pt: class="num">2 }}>
422 <SchoolFormFields form={addForm} setForm={setAddForm} t={t} headmasters={headmasters} />
423 </DialogContent>
424 <DialogActions>
425 <Button onClick={() => setAddOpen(false)}>{t(class="str">'common.cancel')}</Button>
426 <Button variant=class="str">"contained" onClick={handleCreate} disabled={!addForm.name.trim() || saving}>
427 {t(class="str">'common.save')}
428 </Button>
429 </DialogActions>
430 </Dialog>
431
432 {/* Edit School Dialog — includes profile manager */}
433 <Dialog open={editOpen} onClose={() => setEditOpen(false)} fullWidth maxWidth=class="str">"sm">
434 <DialogTitle>{t(class="str">'schools.editSchool')}</DialogTitle>
435 <DialogContent sx={{ display: class="str">'flex', flexDirection: class="str">'column', gap: class="num">2, pt: class="num">2 }}>
436 <SchoolFormFields form={editForm} setForm={setEditForm} t={t} headmasters={headmasters} />
437 {editId && <ProfileManager schoolId={editId} t={t} />}
438 {editId && <ScheduleManager schoolId={editId} t={t} />}
439 </DialogContent>
440 <DialogActions>
441 <Button onClick={() => setEditOpen(false)}>{t(class="str">'common.cancel')}</Button>
442 <Button variant=class="str">"contained" onClick={handleUpdate} disabled={!editForm.name.trim() || saving}>
443 {t(class="str">'common.save')}
444 </Button>
445 </DialogActions>
446 </Dialog>
447
448 {/* Delete Confirmation */}
449 <Dialog open={confirmId !== null} onClose={() => setConfirmId(null)}>
450 <DialogTitle>{t(class="str">'schools.deleteConfirmTitle')}</DialogTitle>
451 <DialogContent>
452 <Typography>{t(class="str">'schools.deleteConfirmMessage', { name: confirmName })}</Typography>
453 </DialogContent>
454 <DialogActions>
455 <Button onClick={() => setConfirmId(null)}>{t(class="str">'common.cancel')}</Button>
456 <Button variant=class="str">"contained" color=class="str">"error" onClick={handleDeleteConfirm}>{t(class="str">'common.delete')}</Button>
457 </DialogActions>
458 </Dialog>
459 </Layout>
460 );
461}

463export default ViewSchools;

Summary
This component provides an administrative interface for managing user accounts within the EduTrack system. It fetches user and school data on mount, then displays them in a sortable and filterable table. Users can create new accounts, edit existing details, or delete records through modal dialogs. The interface supports filtering by role and school, with visual cues for different user types. It handles all CRUD operations via API calls and manages local state for forms and loading indicators. This page serves as the central hub for user administration in the school e-journal.

79function ManageUsers() {
80 const { t } = useTranslation();
81
82 const [users, setUsers] = useState([]);
83 const [schools, setSchools] = useState([]);
84 const [loading, setLoading] = useState(true);
85 const [error, setError] = useState(null);
86 const [saving, setSaving] = useState(false);
87 const [roleFilter, setRoleFilter] = useState(class="str">'ALL');
88 const [schoolFilter, setSchoolFilter] = useState(class="str">'');
89 const [sortBy, setSortBy] = useState(class="str">'id');
90 const [sortDir, setSortDir] = useState(class="str">'asc');
91
92 const [addOpen, setAddOpen] = useState(false);
93 const [addForm, setAddForm] = useState(emptyForm);
94
95 const [editOpen, setEditOpen] = useState(false);
96 const [editId, setEditId] = useState(null);
97 const [editForm, setEditForm] = useState(emptyForm);
98
99 const [confirmId, setConfirmId] = useState(null);
100 const [confirmName, setConfirmName] = useState(class="str">'');
101
102 useEffect(() => {
103 Promise.all([
104 api.get(class="str">'/api/users'),
105 api.get(class="str">'/api/schools'),
106 ])
107 .then(([usersRes, schoolsRes]) => {
108 setUsers(usersRes.data);
109 setSchools(schoolsRes.data);
110 })
111 .catch(() => setError(t(class="str">'users.fetchError')))
112 .finally(() => setLoading(false));
113 }, []);
114
115 const handleSort = (col) => {
116 if (sortBy === col) {
117 setSortDir(d => d === class="str">'asc' ? class="str">'desc' : class="str">'asc');
118 } else {
119 setSortBy(col);
120 setSortDir(class="str">'asc');
121 }
122 };
123
124 const filtered = users
125 .filter(u => roleFilter === class="str">'ALL' || u.role === roleFilter)
126 .filter(u => !schoolFilter || u.schoolId === Number(schoolFilter))
127 .slice()
128 .sort((a, b) => {
129 const av = sortBy === class="str">'id' ? a.id : (a[sortBy] ?? class="str">'').toLowerCase();
130 const bv = sortBy === class="str">'id' ? b.id : (b[sortBy] ?? class="str">'').toLowerCase();
131 if (av < bv) return sortDir === class="str">'asc' ? -class="num">1 : class="num">1;
132 if (av > bv) return sortDir === class="str">'asc' ? class="num">1 : -class="num">1;
133 return class="num">0;
134 });
135
136 const handleCreate = () => {
137 setSaving(true);
138 api.post(class="str">'/api/users', addForm)
139 .then(res => {
140 setUsers(prev => [...prev, res.data]);
141 setAddOpen(false);
142 setAddForm(emptyForm);
143 })
144 .catch(err => setError(err.response?.data?.message || t(class="str">'users.createError')))
145 .finally(() => setSaving(false));
146 };
147
148 const handleRowClick = (user) => {
149 setEditId(user.id);
150 setEditForm({ firstName: user.firstName, lastName: user.lastName, email: user.email, password: class="str">'', role: user.role });
151 setEditOpen(true);
152 };
153
154 const handleUpdate = () => {
155 setSaving(true);
156 api.put(class="str">`/api/users/${editId}`, editForm)
157 .then(res => {
158 setUsers(prev => prev.map(u => u.id === editId ? res.data : u));
159 setEditOpen(false);
160 })
161 .catch(err => setError(err.response?.data?.message || t(class="str">'users.updateError')))
162 .finally(() => setSaving(false));
163 };
164
165 const handleDeleteConfirm = () => {
166 api.delete(class="str">`/api/users/${confirmId}`)
167 .then(() => {
168 setUsers(prev => prev.filter(u => u.id !== confirmId));
169 setConfirmId(null);
170 })
171 .catch(() => {
172 setError(t(class="str">'users.deleteError'));
173 setConfirmId(null);
174 });
175 };
176
177 const isAddValid = addForm.firstName && addForm.lastName && addForm.email && addForm.password && addForm.role;
178 const isEditValid = editForm.firstName && editForm.lastName && editForm.email && editForm.role;
179
180 return (
181 <Layout>
182 <Box sx={{ p: class="num">3 }}>
183 <Box sx={{ display: class="str">'flex', justifyContent: class="str">'space-between', alignItems: class="str">'center', mb: class="num">2 }}>
184 <Typography variant=class="str">"h5">{t(class="str">'users.title')}</Typography>
185 <Button variant=class="str">"contained" startIcon={<AddIcon />} onClick={() => setAddOpen(true)}>
186 {t(class="str">'users.addUser')}
187 </Button>
188 </Box>
189
190 {/* Filters */}
191 <Box sx={{ display: class="str">'flex', gap: class="num">2, mb: class="num">2, flexWrap: class="str">'wrap', alignItems: class="str">'center' }}>
192 <Box sx={{ display: class="str">'flex', gap: class="num">1, flexWrap: class="str">'wrap', alignItems: class="str">'center' }}>
193 {[class="str">'ALL', ...ROLES].map(r => (
194 <Chip
195 key={r}
196 label={t(class="str">`roles.${r}`)}
197 color={roleFilter === r ? class="str">'primary' : class="str">'default'}
198 onClick={() => setRoleFilter(r)}
199 clickable
200 />
201 ))}
202 </Box>
203 <FormControl size=class="str">"small" sx={{ minWidth: class="num">180 }}>
204 <InputLabel sx={{ color: class="str">'black' }}>{t(class="str">'users.filterBySchool')}</InputLabel>
205 <Select
206 value={schoolFilter}
207 onChange={e => setSchoolFilter(e.target.value)}
208 label={t(class="str">'users.filterBySchool')}
209 sx={{ color: class="str">'black' }}
210 >
211 <MenuItem value=class="str">""><em>{t(class="str">'users.allSchools')}</em></MenuItem>
212 {schools.map(s => (
213 <MenuItem key={s.id} value={s.id}>{s.name}</MenuItem>
214 ))}
215 </Select>
216 </FormControl>
217 </Box>
218
219 {error && <Alert severity=class="str">"error" sx={{ mb: class="num">2 }}>{error}</Alert>}
220
221 {loading ? (
222 <Box sx={{ display: class="str">'flex', justifyContent: class="str">'center', mt: class="num">4 }}>
223 <CircularProgress />
224 </Box>
225 ) : (
226 <TableContainer component={Paper}>
227 <Table>
228 <TableHead>
229 <TableRow>
230 <TableCell sx={{ width: class="num">48 }} />
231 {[
232 { id: class="str">'id', label: class="str">'ID' },
233 { id: class="str">'firstName', label: t(class="str">'users.firstName') },
234 { id: class="str">'lastName', label: t(class="str">'users.lastName') },
235 { id: class="str">'email', label: t(class="str">'users.email') },
236 { id: class="str">'role', label: t(class="str">'users.role') },
237 ].map(col => (
238 <TableCell key={col.id} sortDirection={sortBy === col.id ? sortDir : false}>
239 <TableSortLabel
240 active={sortBy === col.id}
241 direction={sortBy === col.id ? sortDir : class="str">'asc'}
242 onClick={() => handleSort(col.id)}
243 >
244 {col.label}
245 </TableSortLabel>
246 </TableCell>
247 ))}
248 <TableCell align=class="str">"center">{t(class="str">'schools.actions')}</TableCell>
249 </TableRow>
250 </TableHead>
251 <TableBody>
252 {filtered.length === class="num">0 ? (
253 <TableRow>
254 <TableCell colSpan={class="num">7} align=class="str">"center">{t(class="str">'users.noUsers')}</TableCell>
255 </TableRow>
256 ) : (
257 filtered.map(user => (
258 <TableRow key={user.id} hover onClick={() => handleRowClick(user)} sx={{ cursor: class="str">'pointer' }}>
259 <TableCell>
260 <UserAvatar userId={user.id} name={class="str">`${user.firstName} ${user.lastName}`} size={class="num">36} />
261 </TableCell>
262 <TableCell>{user.id}</TableCell>
263 <TableCell>{user.firstName}</TableCell>
264 <TableCell>{user.lastName}</TableCell>
265 <TableCell>{user.email}</TableCell>
266 <TableCell>
267 <Chip label={t(class="str">`roles.${user.role}`)} color={ROLE_COLORS[user.role] || class="str">'default'} size=class="str">"small" />
268 </TableCell>
269 <TableCell align=class="str">"center">
270 <IconButton onClick={e => { e.stopPropagation(); setConfirmId(user.id); setConfirmName(class="str">`${user.firstName} ${user.lastName}`); }}>
271 <DeleteIcon sx={{ color: class="str">'red' }} />
272 </IconButton>
273 </TableCell>
274 </TableRow>
275 ))
276 )}
277 </TableBody>
278 </Table>
279 </TableContainer>
280 )}
281 </Box>
282
283 {/* Add User Dialog */}
284 <Dialog open={addOpen} onClose={() => setAddOpen(false)} fullWidth maxWidth=class="str">"sm">
285 <DialogTitle>{t(class="str">'users.addUser')}</DialogTitle>
286 <DialogContent sx={{ display: class="str">'flex', flexDirection: class="str">'column', gap: class="num">2, pt: class="num">2 }}>
287 <UserFormFields form={addForm} setForm={setAddForm} t={t} isEdit={false} />
288 </DialogContent>
289 <DialogActions>
290 <Button onClick={() => setAddOpen(false)}>{t(class="str">'common.cancel')}</Button>
291 <Button variant=class="str">"contained" onClick={handleCreate} disabled={!isAddValid || saving}>
292 {t(class="str">'common.save')}
293 </Button>
294 </DialogActions>
295 </Dialog>
296
297 {/* Edit User Dialog */}
298 <Dialog open={editOpen} onClose={() => setEditOpen(false)} fullWidth maxWidth=class="str">"sm">
299 <DialogTitle>{t(class="str">'users.editUser')}</DialogTitle>
300 <DialogContent sx={{ display: class="str">'flex', flexDirection: class="str">'column', gap: class="num">2, pt: class="num">2 }}>
301 <UserFormFields form={editForm} setForm={setEditForm} t={t} isEdit={true} />
302 </DialogContent>
303 <DialogActions>
304 <Button onClick={() => setEditOpen(false)}>{t(class="str">'common.cancel')}</Button>
305 <Button variant=class="str">"contained" onClick={handleUpdate} disabled={!isEditValid || saving}>
306 {t(class="str">'common.save')}
307 </Button>
308 </DialogActions>
309 </Dialog>
310
311 {/* Delete Confirmation */}
312 <Dialog open={confirmId !== null} onClose={() => setConfirmId(null)}>
313 <DialogTitle>{t(class="str">'users.deleteConfirmTitle')}</DialogTitle>
314 <DialogContent>
315 <Typography>{t(class="str">'users.deleteConfirmMessage', { name: confirmName })}</Typography>
316 </DialogContent>
317 <DialogActions>
318 <Button onClick={() => setConfirmId(null)}>{t(class="str">'common.cancel')}</Button>
319 <Button variant=class="str">"contained" color=class="str">"error" onClick={handleDeleteConfirm}>{t(class="str">'common.delete')}</Button>
320 </DialogActions>
321 </Dialog>
322 </Layout>
323 );
324}
325
326export default ManageUsers;

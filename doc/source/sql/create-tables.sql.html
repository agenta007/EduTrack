Summary
This script initializes the database schema for the EduTrack school e-journal system by defining core entities such as Users, Roles, Schools, Teachers, and Students. It establishes critical relationships through foreign keys, linking teachers to schools, students to classes, and teachers to their qualifications. The structure supports academic operations by creating tables for Subjects, Schedules, Grades, Absences, and Complaints. Each table includes specific constraints and cascading delete rules to maintain data integrity across the system. This foundation enables the application to manage school administration, class assignments, and student performance tracking effectively.

5-- Drop in reverse dependency order
6DROP TABLE IF EXISTS school_schedule_entries CASCADE;
7DROP TABLE IF EXISTS Complaints CASCADE;
8DROP TABLE IF EXISTS Absences CASCADE;
9DROP TABLE IF EXISTS Grades CASCADE;
10DROP TABLE IF EXISTS Schedules CASCADE;
11DROP TABLE IF EXISTS Teacher_Qualifications CASCADE;
12DROP TABLE IF EXISTS Subjects CASCADE;
13DROP TABLE IF EXISTS Students CASCADE;
14DROP TABLE IF EXISTS Teachers CASCADE;
15DROP TABLE IF EXISTS Classes CASCADE;

20-- -------------------------------------------------------------
21-- 1. Roles
22-- -------------------------------------------------------------
23CREATE TABLE Roles (
24 Id SERIAL PRIMARY KEY,
25 Name VARCHAR(50) NOT NULL UNIQUE
26);
27
28-- -------------------------------------------------------------
29-- 2. Users
30-- -------------------------------------------------------------
31CREATE TABLE Users (
32 Id SERIAL PRIMARY KEY,
33 Email VARCHAR(100) NOT NULL UNIQUE,
34 PasswordHash VARCHAR(255) NOT NULL,
35 FirstName VARCHAR(50) NOT NULL,
36 LastName VARCHAR(50) NOT NULL,
37 RoleId INT NOT NULL,
38 Bio VARCHAR(500),
39 FOREIGN KEY (RoleId) REFERENCES Roles(Id) ON DELETE RESTRICT
40);
41
42-- -------------------------------------------------------------
43-- 3. Schools
44-- -------------------------------------------------------------
45CREATE TABLE Schools (
46 Id SERIAL PRIMARY KEY,
47 Name VARCHAR(150) NOT NULL,
48 Address TEXT,
49 DirectorId INT,
50 FOREIGN KEY (DirectorId) REFERENCES Users(Id) ON DELETE SET NULL
51);
52
53-- -------------------------------------------------------------
54-- 4. Classes
55-- -------------------------------------------------------------
56CREATE TABLE Classes (
57 Id SERIAL PRIMARY KEY,
58 Name VARCHAR(20) NOT NULL,
59 SchoolYear VARCHAR(9) NOT NULL,
60 SchoolId INT NOT NULL,
61 FOREIGN KEY (SchoolId) REFERENCES Schools(Id) ON DELETE CASCADE
62);
63
64-- -------------------------------------------------------------
65-- 5. Teachers
66-- -------------------------------------------------------------
67CREATE TABLE Teachers (
68 UserId INT PRIMARY KEY,
69 SchoolId INT,
70 Salary DECIMAL(10,2),
71 FOREIGN KEY (UserId) REFERENCES Users(Id) ON DELETE CASCADE,
72 FOREIGN KEY (SchoolId) REFERENCES Schools(Id) ON DELETE CASCADE
73);
74
75-- -------------------------------------------------------------
76-- 6. Students
77-- -------------------------------------------------------------
78CREATE TABLE Students (
79 UserId INT PRIMARY KEY,
80 SchoolId INT,
81 ClassId INT,
82 ParentId INT,
83 FOREIGN KEY (UserId) REFERENCES Users(Id) ON DELETE CASCADE,
84 FOREIGN KEY (SchoolId) REFERENCES Schools(Id) ON DELETE CASCADE,
85 FOREIGN KEY (ClassId) REFERENCES Classes(Id) ON DELETE SET NULL,
86 FOREIGN KEY (ParentId) REFERENCES Users(Id) ON DELETE SET NULL
87);
88
89-- -------------------------------------------------------------
90-- 8. Subjects
91-- -------------------------------------------------------------
92CREATE TABLE Subjects (
93 Id SERIAL PRIMARY KEY,
94 Name VARCHAR(100) NOT NULL UNIQUE
95);
96
97-- -------------------------------------------------------------
98-- 9. Teacher_Qualifications
99-- -------------------------------------------------------------
100CREATE TABLE Teacher_Qualifications (
101 TeacherId INT NOT NULL,
102 SubjectId INT NOT NULL,
103 PRIMARY KEY (TeacherId, SubjectId),
104 FOREIGN KEY (TeacherId) REFERENCES Teachers(UserId) ON DELETE CASCADE,
105 FOREIGN KEY (SubjectId) REFERENCES Subjects(Id) ON DELETE CASCADE
106);
107
108-- -------------------------------------------------------------
109-- 10. Schedules
110-- -------------------------------------------------------------
111CREATE TABLE Schedules (
112 Id SERIAL PRIMARY KEY,
113 SchoolId INT NOT NULL,
114 ClassId INT NOT NULL,
115 SubjectId INT NOT NULL,
116 TeacherId INT NOT NULL,
117 Term INT NOT NULL CHECK (Term IN (1, 2)),
118 DayOfWeek INT NOT NULL CHECK (DayOfWeek BETWEEN 1 AND 5), -- 1=Mon … 5=Fri
119 StartTime TIME NOT NULL,
120 EndTime TIME NOT NULL,
121 lecture_type VARCHAR(20) NOT NULL DEFAULT 'STANDARD',
122 track_attendance BOOLEAN NOT NULL DEFAULT TRUE,
123 FOREIGN KEY (SchoolId) REFERENCES Schools(Id) ON DELETE CASCADE,
124 FOREIGN KEY (ClassId) REFERENCES Classes(Id) ON DELETE CASCADE,
125 FOREIGN KEY (SubjectId) REFERENCES Subjects(Id) ON DELETE CASCADE,
126 FOREIGN KEY (TeacherId) REFERENCES Teachers(UserId) ON DELETE CASCADE
127);
128
129-- -------------------------------------------------------------
130-- 11. Grades
131-- -------------------------------------------------------------
132CREATE TABLE Grades (
133 Id SERIAL PRIMARY KEY,
134 StudentId INT NOT NULL,
135 ScheduleId INT NOT NULL,
136 Value DECIMAL(3, 2) NOT NULL CHECK (Value IN (2, 2.5, 3, 3.5, 4, 4.5, 5, 5.5, 6)),
137 CreatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
138 FOREIGN KEY (StudentId) REFERENCES Students(UserId) ON DELETE CASCADE,
139 FOREIGN KEY (ScheduleId) REFERENCES Schedules(Id) ON DELETE CASCADE
140);
141
142-- -------------------------------------------------------------
143-- 12. Absences
144-- -------------------------------------------------------------
145CREATE TABLE Absences (
146 Id SERIAL PRIMARY KEY,
147 StudentId INT NOT NULL,
148 ScheduleId INT NOT NULL,
149 Date DATE NOT NULL,
150 IsExcused BOOLEAN DEFAULT FALSE,
151 FOREIGN KEY (StudentId) REFERENCES Students(UserId) ON DELETE CASCADE,
152 FOREIGN KEY (ScheduleId) REFERENCES Schedules(Id) ON DELETE CASCADE
153);
154
155-- -------------------------------------------------------------
156-- 13. Complaints
157-- -------------------------------------------------------------
158CREATE TABLE Complaints (
159 Id SERIAL PRIMARY KEY,
160 StudentId INT NOT NULL,
161 ScheduleId INT NOT NULL,
162 Description VARCHAR(500) NOT NULL,
163 Date DATE NOT NULL,
164 FOREIGN KEY (StudentId) REFERENCES Students(UserId) ON DELETE CASCADE,
165 FOREIGN KEY (ScheduleId) REFERENCES Schedules(Id) ON DELETE CASCADE
166);
167
168-- -------------------------------------------------------------
169-- 14. School_Schedule_Entries (daily timetable + special events)
170-- -------------------------------------------------------------
171CREATE TABLE school_schedule_entries (
172 id SERIAL PRIMARY KEY,
173 schoolid INT NOT NULL,
174 type VARCHAR(20) NOT NULL CHECK (type IN ('LECTURE', 'BREAK', 'SPECIAL_EVENT')),
175 label VARCHAR(100) NOT NULL,
176 start_time TIME NOT NULL,
177 end_time TIME NOT NULL,
178 event_date DATE, -- only for SPECIAL_EVENT
179 sort_order INT NOT NULL DEFAULT 0,
180 FOREIGN KEY (schoolid) REFERENCES Schools(Id) ON DELETE CASCADE
181);

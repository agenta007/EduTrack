63 public SchoolEventDto create(SchoolEventRequest req, UserDetails principal) {
64 User creator = resolveUser(principal);
65
66 EventType type;
67 try {
68 type = EventType.valueOf(req.getType().toUpperCase());
69 } catch (IllegalArgumentException e) {
70 throw new ResponseStatusException(HttpStatus.BAD_REQUEST,

86 SchoolEvent event = SchoolEvent.builder()
87 .title(req.getTitle())
88 .description(req.getDescription())

92 .schoolClass(schoolClass)
93 .createdBy(creator)
94 .build();

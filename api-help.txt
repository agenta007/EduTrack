===============================================================================
 EduTrack / e-journal — API debugging cheat sheet (curl)
===============================================================================
Backend base URL : http://localhost:8080
Auth model       : JWT Bearer access token (sent in `Authorization: Bearer ...`)
                   + HttpOnly `refreshToken` cookie (set by /auth/login).
Generated for    : Spring Boot backend in backend/e-journal

 0. QUICK SETUP
-------------------------------------------------------------------------------

-------------------------------------------------------------------------------
 NOTE: LOCALHOST DEV BYPASS
-------------------------------------------------------------------------------
While `app.security.allow-localhost=true` (default in application.properties),
ANY request from 127.0.0.1 / ::1 that carries NO `Authorization` header is
auto-authenticated as a virtual admin with ALL roles (LocalhostAdminFilter).

  => From this machine, a tokenless curl behaves like an admin:
       curl -s $BASE/api/classes        # works, no token needed

  => To test REAL role-based behaviour (403 / scoping), you MUST send a token
     for a specific user (see section 1). The browser app always sends a token,
     so it is subject to the real @PreAuthorize rules.

Set allow-localhost=false to disable the bypass (staging/production).

-------------------------------------------------------------------------------
 1. AUTH  (/auth/**  — public, no token required)
-------------------------------------------------------------------------------
# --- Login: returns {"accessToken":"..."} and sets the refreshToken cookie ---
curl -s -X POST $BASE/auth/login \
  -H 'Content-Type: application/json' \
  -d '{"email":"h.hristov@school2.bg","password":"password"}'

# --- Login and capture the token into $TOKEN (jq required) ---
export TOKEN=$(curl -s -X POST $BASE/auth/login \
  -H 'Content-Type: application/json' \
  -d '{"email":"h.hristov@school2.bg","password":"password"}' | jq -r .accessToken)
echo "$TOKEN"

# --- Login WITHOUT jq (saves cookie jar too, for refresh/logout) ---
curl -s -c cookies.txt -X POST $BASE/auth/login \
  -H 'Content-Type: application/json' \
  -d '{"email":"p.nikolov@school2.bg","password":"password"}'

# Helper used in every authed example below:  -H "Authorization: Bearer $TOKEN"
export AUTH="Authorization: Bearer $TOKEN"

# --- Refresh: uses the HttpOnly refreshToken cookie, no body ---
curl -s -b cookies.txt -X POST $BASE/auth/refresh

# --- Logout: clears the refresh cookie ---
curl -s -b cookies.txt -X POST $BASE/auth/logout

# --- Quick role switch (re-login as a different demo user) ---
relogin() { export TOKEN=$(curl -s -X POST $BASE/auth/login -H 'Content-Type: application/json' \
  -d "{\"email\":\"$1\",\"password\":\"password\"}" | jq -r .accessToken); export AUTH="Authorization: Bearer $TOKEN"; echo "now: $1"; }
# usage: relogin agenta007@edutrack.bg

 2. CLASSES  (/api/classes)
-------------------------------------------------------------------------------
# List classes — ADMIN: all; HEADMASTER: own school; TEACHER: assigned school
curl -s -H "$AUTH" $BASE/api/classes

# Classes at a school (ADMIN, HEADMASTER)
curl -s -H "$AUTH" $BASE/api/classes/school/2

# One class by id (ADMIN, HEADMASTER, TEACHER, PARENT, STUDENT)
curl -s -H "$AUTH" $BASE/api/classes/1

# Students in a class (ADMIN, HEADMASTER, TEACHER)
curl -s -H "$AUTH" $BASE/api/classes/1/students

-------------------------------------------------------------------------------
 3. GRADES  (/api/grades)   Bulgarian scale 2..6 (2,2.5,...,6)
-------------------------------------------------------------------------------
# All grades for a class (ADMIN, HEADMASTER, TEACHER)
curl -s -H "$AUTH" $BASE/api/grades/class/1

# My grades (STUDENT only)
curl -s -H "$AUTH" $BASE/api/grades/student/me

# Grades for one student (PARENT, ADMIN, HEADMASTER, TEACHER)
curl -s -H "$AUTH" $BASE/api/grades/student/13

# Add a grade (TEACHER, ADMIN, HEADMASTER) -> 201
curl -s -X POST $BASE/api/grades -H "$AUTH" -H 'Content-Type: application/json' \
  -d '{"studentId":13,"scheduleId":1,"value":5.5}'

# Delete a grade (TEACHER, ADMIN, HEADMASTER) -> 204
curl -s -X DELETE $BASE/api/grades/1 -H "$AUTH"

 16. DEBUGGING TIPS
-------------------------------------------------------------------------------
# Show only the HTTP status code:
curl -s -o /dev/null -w "%{http_code}\n" -H "$AUTH" $BASE/api/classes

# Show response headers + body:
curl -s -i -H "$AUTH" $BASE/api/classes

# Verbose (TLS, request/response headers):
curl -sv -H "$AUTH" $BASE/api/classes

# Pretty-print JSON:
curl -s -H "$AUTH" $BASE/api/classes | jq .

# Interpreting status codes here:
#   200/201/204  OK
#   400          bad request / validation / unhandled role in service logic
#   401          missing/expired token (or refresh cookie gone)
#   403          authenticated but role not permitted by @PreAuthorize
#   404          entity not found
#   500          server exception — check backend/e-journal/logs/e-journal.log

# Tail backend logs while testing:
#   tail -f backend/e-journal/logs/e-journal.log         # app + stack traces
#   tail -f backend/e-journal/logs/requests.log          # one line per request
